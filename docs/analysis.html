4 LOESS Plot Function

Code 
loess_plot <- function(data, x, y, x_point, y_point, title, xlab, ylab) {
  ggplot(data, aes(x = {{x}}, y = {{y}})) +
    geom_point(aes(x = {{x_point}}, y = {{y_point}}),
               size = 6, color = "black", shape = 19) +
    geom_smooth(method = "loess", se = TRUE, span = 0.75, color = "blue", linewidth = 2) +
    scale_x_continuous(breaks = seq(0, 100, by = 10)) +
    scale_y_continuous(labels = percent_format(accuracy = 1)) +
    labs(title = title, x = xlab, y = ylab) +
    theme_minimal() +
    theme(
      axis.title.x  = element_text(size = 20),
      axis.title.y  = element_text(size = 20),
      axis.text.x   = element_text(size = 16),
      axis.text.y   = element_text(size = 16),
      plot.title     = element_text(size = 24, face = "bold")
    ) +
    annotate("rect", xmin = 1, xmax = 25, ymin = -Inf, ymax = Inf,
             alpha = 0.2, fill = "yellow")
}
