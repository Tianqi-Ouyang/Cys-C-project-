loess_plot <- function(data, x, y, x_point, y_point, title, xlab, ylab,
                       vlines = NULL) {
  p <- ggplot(data, aes(x = {{x}}, y = {{y}})) +
    geom_point(aes(x = {{x_point}}, y = {{y_point}}),
               size = 4, color = "black", shape = 19) +
    geom_smooth(method = "loess", se = TRUE, span = 0.75, color = "blue", linewidth = 1.5) +
    scale_y_continuous(labels = percent_format(accuracy = 1)) +
    labs(title = title, x = xlab, y = ylab) +
    theme_minimal(base_size = 12) +
    theme(
      axis.title.x  = element_text(size = 11),
      axis.title.y  = element_text(size = 11),
      axis.text.x   = element_text(size = 10),
      axis.text.y   = element_text(size = 10),
      plot.title     = element_text(size = 13, face = "bold")
    )
  if (!is.null(vlines)) {
    p <- p +
      geom_vline(data = vlines, aes(xintercept = x),
                 linetype = "dashed", color = "red", linewidth = 0.7) +
      geom_text(data = vlines, aes(x = x, y = Inf, label = label),
                vjust = 1.5, hjust = -0.1, size = 3.2, color = "red")
  }
  p
}

loess_plot_raw <- function(data, x, y, title, xlab, ylab, vlines = NULL) {
  p <- ggplot(data, aes(x = {{x}}, y = {{y}})) +
    geom_smooth(method = "loess", se = TRUE, span = 0.75, color = "blue", linewidth = 1.5) +
    scale_y_continuous(labels = percent_format(accuracy = 1)) +
    labs(title = title, x = xlab, y = ylab) +
    theme_minimal(base_size = 12) +
    theme(
      axis.title.x  = element_text(size = 11),
      axis.title.y  = element_text(size = 11),
      axis.text.x   = element_text(size = 10),
      axis.text.y   = element_text(size = 10),
      plot.title     = element_text(size = 13, face = "bold")
    )
  if (!is.null(vlines)) {
    p <- p +
      geom_vline(data = vlines, aes(xintercept = x),
                 linetype = "dashed", color = "red", linewidth = 0.7) +
      geom_text(data = vlines, aes(x = x, y = Inf, label = label),
                vjust = 1.5, hjust = -0.1, size = 3.2, color = "red")
  }
  p
}

# Subset to carboplatin patients with AUC < 4
fm_lt4 <- final_master %>%
  filter(carbo_AUC_cat == "<4")

# Recompute percentile and decile within this subset
fm_lt4 <- fm_lt4 %>%
  mutate(
    dd_percentile_lt4 = ntile(dose_discrepancy, 100),
    dd_decile_lt4     = ntile(dose_discrepancy, 10)
  ) %>%
  group_by(dd_decile_lt4) %>%
  mutate(
    anemia_2_rate_decile_lt4             = mean(Anemia_g2_90days, na.rm = TRUE),
    anemia_3_rate_decile_lt4             = mean(Anemia_g3_90days, na.rm = TRUE),
    throm_2_rate_decile_lt4              = mean(Throm_g2_90days, na.rm = TRUE),
    throm_3_rate_decile_lt4              = mean(Throm_g3_90days, na.rm = TRUE),
    aki_2_rate_decile_lt4                = mean(AKI_g2_90days, na.rm = TRUE),
    aki_3_rate_decile_lt4                = mean(AKI_g3_90days, na.rm = TRUE),
    hos_rate_decile_lt4                  = mean(hosp_90, na.rm = TRUE),
    platinum_related_hos_rate_decile_lt4 = mean(platinum_related_hosp, na.rm = TRUE),
    death_rate_decile_lt4                = mean(death_90, na.rm = TRUE)
  ) %>%
  ungroup() %>%
  mutate(dd_decile_lt4 = as.numeric(dd_decile_lt4))
Code 
# Median dose discrepancy at P60 and P70 within the AUC < 4 subset
median_dd_p60_lt4 <- median(
  fm_lt4$dose_discrepancy[fm_lt4$dd_percentile_lt4 == 60], na.rm = TRUE)
median_dd_p70_lt4 <- median(
  fm_lt4$dose_discrepancy[fm_lt4$dd_percentile_lt4 == 70], na.rm = TRUE)

pct_vlines_lt4 <- data.frame(
  x     = c(60, 70),
  label = c(paste0("median: ", round(median_dd_p60_lt4, 0), ""),
            paste0("median: ", round(median_dd_p70_lt4, 0), ""))
)
