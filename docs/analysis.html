dose_long <- final_master %>%
  select(record_id, carbo_dose_redcap, carbo_dose_cre_cys) %>%
  pivot_longer(
    cols = c(carbo_dose_redcap, carbo_dose_cre_cys),
    names_to = "dose_type", values_to = "dose_mg"

  filter(!is.na(dose_mg)) %>%
  mutate(dose_type = case_match(dose_type,
    "carbo_dose_redcap"  ~ "Actual (REDCap)",
    "carbo_dose_cre_cys" ~ "Predicted (CKD-EPI Cr-Cys)"
  ))

ggplot(dose_long, aes(x = dose_mg, fill = dose_type)) +
  geom_histogram(alpha = 0.5, bins = 40, position = "identity", color = "black") +
  scale_fill_manual(values = c("Actual (REDCap)" = "#E31A1C",
                                "Predicted (CKD-EPI Cr-Cys)" = "#1F78B4")) +
  labs(title = "Carboplatin Dose Distribution",
       x = "Dose (mg)", y = "Number of Patients", fill = "Dose Type") +
  theme_minimal(base_size = 14)

loess_plot <- function(data, x, y, x_point, y_point, title, xlab, ylab,
                       vlines = NULL) {
  p <- ggplot(data, aes(x = {{x}}, y = {{y}})) +
    geom_point(aes(x = {{x_point}}, y = {{y_point}}),
               size = 4, color = "black", shape = 19) +
    geom_smooth(method = "loess", se = TRUE, span = 0.75, color = "blue", linewidth = 1.5) +
    scale_y_continuous(labels = percent_format(accuracy = 1)) +
    labs(title = title, x = xlab, y = ylab) +
    theme_minimal(base_size = 12) +
    theme(
      axis.title.x  = element_text(size = 11),
      axis.title.y  = element_text(size = 11),
      axis.text.x   = element_text(size = 10),
      axis.text.y   = element_text(size = 10),
      plot.title     = element_text(size = 13, face = "bold")
    )
  if (!is.null(vlines)) {
    p <- p +
      geom_vline(data = vlines, aes(xintercept = x),
                 linetype = "dashed", color = "red", linewidth = 0.7) +
      geom_text(data = vlines, aes(x = x, y = Inf, label = label),
                vjust = 1.5, hjust = -0.1, size = 3.2, color = "red")
  }
  p
}

loess_plot_raw <- function(data, x, y, title, xlab, ylab) {
  ggplot(data, aes(x = {{x}}, y = {{y}})) +
    geom_smooth(method = "loess", se = TRUE, span = 0.75, color = "blue", linewidth = 1.5) +
    scale_y_continuous(labels = percent_format(accuracy = 1)) +
    labs(title = title, x = xlab, y = ylab) +
    theme_minimal(base_size = 12) +
    theme(
      axis.title.x  = element_text(size = 11),
      axis.title.y  = element_text(size = 11),
      axis.text.x   = element_text(size = 10),
      axis.text.y   = element_text(size = 10),
      plot.title     = element_text(size = 13, face = "bold")
    )
}
