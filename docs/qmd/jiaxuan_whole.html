# fig.process runs in the chunk eval context (so last_plot() is fresh) and
# writes an editable .pptx alongside the saved figure.
knitr::opts_chunk$set(fig.process = function(filename) {
  label <- knitr::opts_current$get("label")
  if (is.null(label)) return(filename)
  p <- tryCatch(ggplot2::last_plot(), error = function(e) NULL)
  if (is.null(p) || !inherits(p, "ggplot")) return(filename)
  pptx_path <- file.path(pptx_dir, paste0(label, ".pptx"))
  tryCatch({
    doc <- officer::read_pptx()
    doc <- officer::add_slide(doc, layout = "Title and Content",
                              master = "Office Theme")
    doc <- officer::ph_with(doc, rvg::dml(ggobj = p),
                            location = officer::ph_location_type(type = "body"))
    print(doc, target = pptx_path)
    .pptx_written[[label]] <- paste0(pptx_rel_prefix, label, ".pptx")
  }, error = function(e) NULL)
  filename
})

# A "chunk" hook fires after the whole chunk is rendered: append the link.

  default_chunk_hook <- knitr::knit_hooks$get("chunk")

  })

# Create eGFR ratio: CKD-EPI Cr-Cys (unindexed) / Cockcroft-Gault
# Also create per-10 mL/min rescaled versions of cockcroft and CKD-EPI for
# reporting (HRs per 1 mL/min are clinically uninterpretable; convention in
# nephro-onc is per 10 mL/min — bio_agent confirmed). Adjusters keep raw scale.
dat <- dat %>%
  mutate(
    # Ratio uses UNCAPPED eGFR
    egfr_ratio    = ckd_epi_gfr_cre_cys_unindex / cockcroft,
    egfr_ratio_per10 = -egfr_ratio * 10,  # per 0.1-unit *decrease* reporting scale
    egfr_discrepancy = ckd_epi_gfr_cre_cys_unindex - cockcroft,
    # per-10-mL/min DECLINE reporting scale for eGFR discrepancy: HR for this
    # variable reads as the multiplicative change in subHR per 10 mL/min lower
    # discrepancy (i.e., when CKD-EPI Cr-Cys is 10 mL/min farther below CG).
    egfr_disc_per10dec = -egfr_discrepancy / 10,
    # Sensitivity-analysis eGFR ratio (Section 11):
    # numerator = CKD-EPI 2012 cystatin C only (baseline_cys_c_egfr_unindexed)
    # denominator = CKD-EPI 2021 race-free creatinine only (baseline_cre_egfr_unindexed)
    # (Both unindexed; the ratio itself is invariant to whether indexed or unindexed
    # because BSA cancels.) per10 below is the per-0.1 DECLINE reporting scale.
    egfr_ratio_sens       = baseline_cys_c_egfr_unindexed / baseline_cre_egfr_unindexed,
    egfr_ratio_sens_per10 = -egfr_ratio_sens * 10,
    cockcroft_per10 = cockcroft / 10,
    ckd_epi_per10   = ckd_epi_gfr_cre_cys_unindex / 10
  )

.adj_set <- function(pred_var, pre_labs, egfr_adj) {
  adj <- c("age","sex","score","ecog_score_cat2",
           "cancer_type_cat_7","tumor_stage","platin_group","high_dose",
           "Paclitaxel","Pemetrexed","Gemcitabine","ici",
           "bmi","smoking","steroids","thyroid")
  # Drop platin_group when predictor requires dose data (carbo-only after drop_na)
  if (grepl("dose_discrep", pred_var)) adj <- setdiff(adj, "platin_group")
  if (egfr_adj == "cockcroft") adj <- c(adj, "cockcroft")
  if (egfr_adj == "ckd_epi")   adj <- c(adj, "ckd_epi_gfr_cre_cys_unindex")
  if (!is.null(pre_labs)) adj <- c(adj, pre_labs)
  adj
}

# fg_model: returns univariable + multivariable merged Fine-Gray table.
# If docx_name is supplied, also saves the table as a .docx in docx_dir.
fg_model <- function(time_var, status_var, pred_var, pre_labs = NULL,
                     egfr_adj = c("none", "cockcroft", "ckd_epi"),
                     docx_name = NULL) {
  egfr_adj <- match.arg(egfr_adj)
  adj <- .adj_set(pred_var, pre_labs, egfr_adj)
  all_vars <- c(pred_var, adj)

  # Multivariable
  fmla_mv <- as.formula(paste0(
    "Surv(", time_var, ", ", status_var, ") ~ ",
    paste(all_vars, collapse = " + ")
  ))
  tbl_mv <- tidycmprsk::crr(fmla_mv, failcode = 1, data = dat) %>%
    tbl_regression(exp = TRUE,
                   pvalue_fun = function(x) style_pvalue(x, digits = 3)) %>%
    add_n()

  # Univariable: one Fine-Gray fit per variable
  y_call <- str2lang(sprintf("Surv(%s, %s)", time_var, status_var))
  tbl_uv <- rlang::inject(
    tbl_uvregression(
      data = dat,
      method = tidycmprsk::crr,
      y = !!y_call,
      include = dplyr::all_of(all_vars),
      exponentiate = TRUE,
      pvalue_fun = function(x) style_pvalue(x, digits = 3),
      method.args = list(failcode = 1)
    )
  )

  out <- tbl_merge(
    tbls = list(tbl_uv, tbl_mv),
    tab_spanner = c("**Univariable**", "**Multivariable**")
  )

  if (!is.null(docx_name)) save_tbl_docx(out, docx_name)
  out
}

# cox_model: same signature/output for the 90-day death analysis (Cox PH).
cox_model <- function(time_var, status_var, pred_var, pre_labs = NULL,
                      egfr_adj = c("none", "cockcroft", "ckd_epi"),
                      docx_name = NULL) {
  egfr_adj <- match.arg(egfr_adj)
  adj <- .adj_set(pred_var, pre_labs, egfr_adj)
  all_vars <- c(pred_var, adj)

  fmla_mv <- as.formula(paste0(
    "Surv(", time_var, ", ", status_var, ") ~ ",
    paste(all_vars, collapse = " + ")
  ))
  tbl_mv <- survival::coxph(fmla_mv, data = dat) %>%
    tbl_regression(exp = TRUE,
                   pvalue_fun = function(x) style_pvalue(x, digits = 3)) %>%
    add_n()

  y_call <- str2lang(sprintf("Surv(%s, %s)", time_var, status_var))
  tbl_uv <- rlang::inject(
    tbl_uvregression(
      data = dat,
      method = survival::coxph,
      y = !!y_call,
      include = dplyr::all_of(all_vars),
      exponentiate = TRUE,
      pvalue_fun = function(x) style_pvalue(x, digits = 3)
    )
  )

  out <- tbl_merge(
    tbls = list(tbl_uv, tbl_mv),
    tab_spanner = c("**Univariable**", "**Multivariable**")
  )

  if (!is.null(docx_name)) save_tbl_docx(out, docx_name)
  out
}

csh_model <- function(time_var, status_var, pred_var, pre_labs = NULL) {
  adj <- c("age","sex","score","ecog_score_cat2",
           "cancer_type_cat_7","tumor_stage","platin_group","high_dose",
           "Paclitaxel","Pemetrexed","Gemcitabine","ici",
           "bmi","smoking","steroids","thyroid","cockcroft")
  if (grepl("dose_discrep", pred_var)) adj <- setdiff(adj, "platin_group")
  if (!is.null(pre_labs)) adj <- c(adj, pre_labs)
  # Cause-specific: treat death (status == 2) as censored
  d <- dat
  d$csh_status <- ifelse(d[[status_var]] == 1, 1L, 0L)
  fmla <- as.formula(paste0(
    "Surv(", time_var, ", csh_status) ~ ",
    paste(c(pred_var, adj), collapse = " + ")
  ))
  survival::coxph(fmla, data = d) %>%
    tbl_regression(exp = TRUE,
                   pvalue_fun = function(x) style_pvalue(x, digits = 3)) %>%
    add_n()
}

NoteWhy anchor at the median (subHR = 1 at the reference)

Three reasons to anchor at the median:

  1. Interpretability: “A patient with eGFR ratio of 0.8 has ~1.5× the subHR compared to a typical patient (at the median).”
  2. Stability: The median sits in the densest part of the data, so estimates near it have the tightest confidence intervals. The CIs fan out as you move away — you can see this in the plots.
  3. No arbitrary choice: If we picked 1.0 arbitrarily, the plot would look almost identical here (because median ≈ 1), but in cohorts where the median is 0.9 or 1.1, picking 1.0 would either shift or distort the curve without changing the actual model.

plot_subhr_linear_ratio <- function(time_var, status_var, extra_adj = NULL,
                                    outcome_label, x_axis = c("ratio","percentile"),
                                    n_bonf = 6) {
  x_axis <- match.arg(x_axis)
  base_adj <- c("age","sex","score","ecog_score_cat2","cockcroft",
                "cancer_type_cat_7","tumor_stage","platin_group","high_dose",
                "Paclitaxel","Pemetrexed","Gemcitabine","ici",
                "bmi","smoking","steroids","thyroid")
  keep <- c("egfr_ratio", time_var, status_var, base_adj, extra_adj)
  d <- dat %>% dplyr::select(all_of(keep)) %>% tidyr::drop_na()

  adj_df <- d[, c(base_adj, extra_adj), drop = FALSE]
  X_adj <- model.matrix(~ ., data = adj_df)[, -1, drop = FALSE]
  X <- cbind(egfr_ratio = d$egfr_ratio, X_adj)

  stat_num <- suppressWarnings(as.integer(as.character(d[[status_var]])))
  time_num <- as.numeric(d[[time_var]])

  fit <- cmprsk::crr(ftime = time_num, fstatus = stat_num,
                     cov1 = X, failcode = 1)

  beta   <- fit$coef[1]
  V_beta <- fit$var[1, 1]
  se_b   <- sqrt(V_beta)

  # subHR per 0.1 ratio decline and per IQR decline (delta-method CIs)
  q1    <- as.numeric(quantile(d$egfr_ratio, 0.25, na.rm = TRUE))
  q3    <- as.numeric(quantile(d$egfr_ratio, 0.75, na.rm = TRUE))
  iqr_w <- q3 - q1
  hr01  <- exp(-0.1 * beta);     hr01_l <- exp(-0.1 * beta - 1.96 * 0.1 * se_b);   hr01_u <- exp(-0.1 * beta + 1.96 * 0.1 * se_b)
  hri   <- exp(-iqr_w * beta);   hri_l  <- exp(-iqr_w * beta - 1.96 * abs(iqr_w) * se_b); hri_u <- exp(-iqr_w * beta + 1.96 * abs(iqr_w) * se_b)
  pval  <- 2 * (1 - pnorm(abs(beta / se_b)))
  pval_bonf <- min(pval * n_bonf, 1)
  p_shown <- if (n_bonf > 1) pval_bonf else pval
  p_lbl   <- if (n_bonf > 1) sprintf("p_Bonf = %s (n = %d)", ifelse(p_shown < 0.001, "<0.001", sprintf("%.3f", p_shown)), n_bonf) else sprintf("p = %s", ifelse(p_shown < 0.001, "<0.001", sprintf("%.3f", p_shown)))
  subtitle_str <- sprintf(
    "subHR per 0.1 ratio decline: %.2f (%.2f-%.2f), %s  |  per IQR decline (Δ = %.2f): %.2f (%.2f-%.2f)  |  IQR: [%.2f, %.2f]",
    hr01, hr01_l, hr01_u, p_lbl, iqr_w, hri, hri_l, hri_u, q1, q3)

  ref_x  <- median(d$egfr_ratio, na.rm = TRUE)
  grid_x <- seq(quantile(d$egfr_ratio, 0.025, na.rm = TRUE),
                quantile(d$egfr_ratio, 0.975, na.rm = TRUE),
                length.out = 100)

  log_hr <- beta * (grid_x - ref_x)
  se     <- sqrt((grid_x - ref_x)^2 * V_beta)

  if (x_axis == "percentile") {
    ecdf_fn      <- ecdf(d$egfr_ratio)
    grid_display <- ecdf_fn(grid_x) * 100
    ref_display  <- ecdf_fn(ref_x)  * 100
    xlab <- "Percentile of eGFR Ratio"
  } else {
    grid_display <- grid_x
    ref_display  <- ref_x
    xlab <- "CKD-EPI Cr-Cys / CG ratio"
  }

  df_plot <- tibble(
    x     = grid_display,
    subhr = exp(log_hr),
    lcl   = exp(log_hr - 1.96 * se),
    ucl   = exp(log_hr + 1.96 * se)
  )

  p <- ggplot(df_plot, aes(x = x, y = subhr)) +
    geom_ribbon(aes(ymin = lcl, ymax = ucl), alpha = 0.2, fill = "darkorange") +
    geom_line(color = "darkorange", linewidth = 1) +
    geom_hline(yintercept = 1, linetype = "dashed", color = "grey50") +
    geom_vline(xintercept = ref_display, linetype = "dotted", color = "grey40") +
    labs(x = xlab, y = "subHR (95% CI)",
         title = outcome_label,
         subtitle = subtitle_str) +
    theme_minimal(base_size = 11) +
    theme(plot.subtitle = element_text(size = 9, color = "grey25"))

  if (x_axis == "ratio") {
    p <- p + coord_cartesian(xlim = c(0.65, 1.45), ylim = c(0, 4))
  } else {
    p <- p + coord_cartesian(xlim = c(0.65, 1.45), ylim = c(0, 4))
  }
  p
}

7.0 Summary table: Fine-Gray subHR per 10 mL/min decline in eGFR discrepancy

One row per outcome. Multivariable Fine-Gray model fit with egfr_disc_per10dec = -egfr_discrepancy / 10 (positive when CKD-EPI Cr-Cys is below CG) plus the same adjusters used by the curves below. fg_disc_row <- function(time_var, status_var, label, pre_labs = NULL) {

  b <- unname(fit$coef[1]); se <- sqrt(fit$var[1, 1])

    N       = nrow(d),
    Events  = sum(stat_num == 1, na.rm = TRUE),
    `(sub)HR (95% CI)` = sprintf("%.2f (%.2f, %.2f)", hr, lcl, ucl),

}

plot_subhr_linear_disc <- function(time_var, status_var, extra_adj = NULL,
                                   outcome_label, n_bonf = 6) {
  base_adj <- c("age","sex","score","ecog_score_cat2",
                "cancer_type_cat_7","tumor_stage","platin_group","high_dose",
                "Paclitaxel","Pemetrexed","Gemcitabine","ici",
                "bmi","smoking","steroids","thyroid")
  keep <- c("egfr_discrepancy", time_var, status_var, base_adj, extra_adj)
  d <- dat %>% dplyr::select(all_of(keep)) %>% tidyr::drop_na()

  adj_df <- d[, c(base_adj, extra_adj), drop = FALSE]
  X_adj <- model.matrix(~ ., data = adj_df)[, -1, drop = FALSE]
  X <- cbind(egfr_discrepancy = d$egfr_discrepancy, X_adj)

  stat_num <- suppressWarnings(as.integer(as.character(d[[status_var]])))
  time_num <- as.numeric(d[[time_var]])

  fit <- cmprsk::crr(ftime = time_num, fstatus = stat_num,
                     cov1 = X, failcode = 1)

  beta   <- fit$coef[1]
  V_beta <- fit$var[1, 1]
  se_b   <- sqrt(V_beta)

  # subHR summaries: per 1 mL/min, and per IQR (Q3 - Q1) of egfr_discrepancy
  q1    <- as.numeric(quantile(d$egfr_discrepancy, 0.25, na.rm = TRUE))
  q3    <- as.numeric(quantile(d$egfr_discrepancy, 0.75, na.rm = TRUE))
  iqr_w <- q3 - q1
  # per 10 mL/min DECLINE: HR = exp(-10 * beta); SE on log scale = 10 * se_b
  hr10  <- exp(-10 * beta);       hr10_l <- exp(-10 * beta - 1.96 * 10 * se_b);     hr10_u <- exp(-10 * beta + 1.96 * 10 * se_b)
  # per IQR DECLINE: HR = exp(-iqr_w * beta); SE on log scale = |iqr_w| * se_b
  hri   <- exp(-iqr_w * beta);    hri_l <- exp(-iqr_w * beta - 1.96 * abs(iqr_w) * se_b); hri_u <- exp(-iqr_w * beta + 1.96 * abs(iqr_w) * se_b)
  pval  <- 2 * (1 - pnorm(abs(beta / se_b)))
  pval_bonf <- min(pval * n_bonf, 1)
  p_shown <- if (n_bonf > 1) pval_bonf else pval
  p_lbl   <- if (n_bonf > 1) sprintf("p_Bonf = %s (n = %d)", ifelse(p_shown < 0.001, "<0.001", sprintf("%.3f", p_shown)), n_bonf) else sprintf("p = %s", ifelse(p_shown < 0.001, "<0.001", sprintf("%.3f", p_shown)))
  subtitle_str <- sprintf(
    "subHR per 10 mL/min decline: %.2f (%.2f-%.2f), %s  |  per IQR decline (Δ = %.1f): %.2f (%.2f-%.2f)  |  IQR: [%.1f, %.1f]",
    hr10, hr10_l, hr10_u, p_lbl, iqr_w, hri, hri_l, hri_u, q1, q3)

  ref_x  <- median(d$egfr_discrepancy, na.rm = TRUE)
  grid_x <- seq(quantile(d$egfr_discrepancy, 0.025, na.rm = TRUE),
                quantile(d$egfr_discrepancy, 0.975, na.rm = TRUE),
                length.out = 100)

  log_hr <- beta * (grid_x - ref_x)
  se     <- sqrt((grid_x - ref_x)^2 * V_beta)

  df_plot <- tibble(
    x     = grid_x,
    subhr = exp(log_hr),
    lcl   = exp(log_hr - 1.96 * se),
    ucl   = exp(log_hr + 1.96 * se)
  )

  ggplot(df_plot, aes(x = x, y = subhr)) +
    geom_ribbon(aes(ymin = lcl, ymax = ucl), alpha = 0.2, fill = "darkorange") +
    geom_line(color = "darkorange", linewidth = 1) +
    geom_hline(yintercept = 1, linetype = "dashed", color = "grey50") +
    geom_vline(xintercept = ref_x, linetype = "dotted", color = "grey40") +
    geom_vline(xintercept = c(q1, q3), linetype = "dashed",
               color = "steelblue", alpha = 0.7) +
    labs(x = "eGFR Discrepancy (mL/min)",
         y = "subHR (95% CI)",
         title = outcome_label,
         subtitle = subtitle_str) +
    coord_cartesian(ylim = c(0, 4)) +
    theme_minimal(base_size = 11) +
    theme(plot.subtitle = element_text(size = 9, color = "grey25"))
}

fg_sens_row <- function(time_var, status_var, label, pre_labs = NULL) {
  adj <- .sens_adj_set(pre_labs)
  d <- dat %>%
    dplyr::select(all_of(c("egfr_ratio_sens_per10", time_var, status_var, adj))) %>%
    tidyr::drop_na()
  adj_df <- d[, adj, drop = FALSE]
  X_adj  <- model.matrix(~ ., data = adj_df)[, -1, drop = FALSE]

  stat_num <- suppressWarnings(as.integer(as.character(d[[status_var]])))
  time_num <- as.numeric(d[[time_var]])
  fit <- cmprsk::crr(time_num, stat_num, cov1 = X, failcode = 1)
  b <- unname(fit$coef[1]); se <- sqrt(fit$var[1, 1])
  hr <- exp(b); lcl <- exp(b - 1.96 * se); ucl <- exp(b + 1.96 * se)
  pv <- 2 * (1 - pnorm(abs(b / se)))
  tibble(
    Outcome = label,
    Model   = "Fine-Gray",
    N       = nrow(d),
    Events  = sum(stat_num == 1, na.rm = TRUE),
    `(sub)HR (95% CI)` = sprintf("%.2f (%.2f, %.2f)", hr, lcl, ucl),
    `p-value` = ifelse(pv < 0.001, "<0.001", sprintf("%.3f", pv))
  )
}

plot_subhr_linear_sens <- function(time_var, status_var, extra_adj = NULL,
                                   outcome_label, n_bonf = 6) {
  base_adj <- c("age","sex","score","ecog_score_cat2","cockcroft",
                "cancer_type_cat_7","tumor_stage","platin_group","high_dose",
                "Paclitaxel","Pemetrexed","Gemcitabine","ici",
                "bmi","smoking","steroids","thyroid")
  keep <- c("egfr_ratio_sens", time_var, status_var, base_adj, extra_adj)
  d <- dat %>% dplyr::select(all_of(keep)) %>% tidyr::drop_na()

  adj_df <- d[, c(base_adj, extra_adj), drop = FALSE]
  X_adj <- model.matrix(~ ., data = adj_df)[, -1, drop = FALSE]
  X <- cbind(egfr_ratio_sens = d$egfr_ratio_sens, X_adj)

  stat_num <- suppressWarnings(as.integer(as.character(d[[status_var]])))
  time_num <- as.numeric(d[[time_var]])

  fit <- cmprsk::crr(ftime = time_num, fstatus = stat_num,
                     cov1 = X, failcode = 1)

  beta   <- fit$coef[1]
  V_beta <- fit$var[1, 1]
  se_b   <- sqrt(V_beta)

  q1    <- as.numeric(quantile(d$egfr_ratio_sens, 0.25, na.rm = TRUE))
  q3    <- as.numeric(quantile(d$egfr_ratio_sens, 0.75, na.rm = TRUE))
  iqr_w <- q3 - q1
  hr01  <- exp(-0.1 * beta);     hr01_l <- exp(-0.1 * beta - 1.96 * 0.1 * se_b);   hr01_u <- exp(-0.1 * beta + 1.96 * 0.1 * se_b)
  hri   <- exp(-iqr_w * beta);   hri_l  <- exp(-iqr_w * beta - 1.96 * abs(iqr_w) * se_b); hri_u <- exp(-iqr_w * beta + 1.96 * abs(iqr_w) * se_b)
  pval  <- 2 * (1 - pnorm(abs(beta / se_b)))
  pval_bonf <- min(pval * n_bonf, 1)
  p_shown <- if (n_bonf > 1) pval_bonf else pval
  p_lbl   <- if (n_bonf > 1) sprintf("p_Bonf = %s (n = %d)", ifelse(p_shown < 0.001, "<0.001", sprintf("%.3f", p_shown)), n_bonf) else sprintf("p = %s", ifelse(p_shown < 0.001, "<0.001", sprintf("%.3f", p_shown)))
  subtitle_str <- sprintf(
    "subHR per 0.1 ratio decline: %.2f (%.2f-%.2f), %s  |  per IQR decline (Δ = %.2f): %.2f (%.2f-%.2f)  |  IQR: [%.2f, %.2f]",
    hr01, hr01_l, hr01_u, p_lbl, iqr_w, hri, hri_l, hri_u, q1, q3)

  ref_x  <- median(d$egfr_ratio_sens, na.rm = TRUE)
  grid_x <- seq(quantile(d$egfr_ratio_sens, 0.025, na.rm = TRUE),
                quantile(d$egfr_ratio_sens, 0.975, na.rm = TRUE),
                length.out = 100)

  log_hr <- beta * (grid_x - ref_x)
  se     <- sqrt((grid_x - ref_x)^2 * V_beta)

  df_plot <- tibble(
    x     = grid_x,
    subhr = exp(log_hr),
    lcl   = exp(log_hr - 1.96 * se),
    ucl   = exp(log_hr + 1.96 * se)
  )

  ggplot(df_plot, aes(x = x, y = subhr)) +
    geom_ribbon(aes(ymin = lcl, ymax = ucl), alpha = 0.2, fill = "darkorange") +
    geom_line(color = "darkorange", linewidth = 1) +
    geom_hline(yintercept = 1, linetype = "dashed", color = "grey50") +
    geom_vline(xintercept = ref_x, linetype = "dotted", color = "grey40") +
    geom_vline(xintercept = c(q1, q3), linetype = "dashed",
               color = "steelblue", alpha = 0.7) +
    labs(x = "Sensitivity eGFR Ratio (CKD-EPI 2012 CysC / CKD-EPI 2021 Cr)",
         y = "subHR (95% CI)",
         title = outcome_label,
         subtitle = subtitle_str) +
    coord_cartesian(ylim = c(0, 4)) +
    theme_minimal(base_size = 11) +
    theme(plot.subtitle = element_text(size = 9, color = "grey25"))
}
