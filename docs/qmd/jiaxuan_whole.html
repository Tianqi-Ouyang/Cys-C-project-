loess_plot <- function(data, x, y, title, xlab, ylab) {
  ggplot(data, aes(x = {{x}}, y = {{y}})) +
    geom_smooth(method = "loess", se = TRUE, span = 0.75,
                color = "blue", linewidth = 1.5) +
    scale_y_continuous(labels = percent_format(accuracy = 1)) +
    labs(title = title, x = xlab, y = ylab) +
    theme_minimal(base_size = 12) +
    theme(
      axis.title.x  = element_text(size = 11),
      axis.title.y  = element_text(size = 11),
      axis.text.x   = element_text(size = 10),
      axis.text.y   = element_text(size = 10),
      plot.title     = element_text(size = 13, face = "bold")
    )
}

5 Model Diagnostics

5.1 Events per Variable (EPV)

Models with EPV < 10 are at risk of instability; EPV < 5 should not be reported without penalisation.

Code 
outcomes <- c("anemia_2", "anemia_3", "throm_2", "throm_3",
              "aki_2", "aki_3", "grade_2_aes", "grade_3_aes")

epv_table <- purrr::map_dfr(outcomes, function(out) {

  status_col <- paste0("cmp_status_", out)
  n_pred <- case_when(
    grepl("anemia", out)  ~ 15L,
    grepl("throm", out)   ~ 15L,
    grepl("aki", out)     ~ 15L,
    TRUE                  ~ 16L
  )
  tibble(
    outcome      = out,
    N            = sum(!is.na(dat[[status_col]])),
    events       = sum(dat[[status_col]] == 1, na.rm = TRUE),
    deaths       = sum(dat[[status_col]] == 2, na.rm = TRUE),
    n_predictors = n_pred,
    EPV          = round(events / n_pred, 1)
  )
})

5.2 Collinearity — VIF (eGFR Ratio Model)

VIF > 5 = concerning; VIF > 10 = problematic.

Code 
library(car)

vif_dat <- dat %>%
  select(grade_2_aes, egfr_ratio, age, sex, bmi, smoking, steroids, thyroid,
         tumor_stage, ecog_score_cat2, Paclitaxel, Pemetrexed, Gemcitabine,
         ici, score, pre_HGB_45days, pre_PLT_45days) %>%
  drop_na()

vif_mod <- glm(grade_2_aes ~ egfr_ratio + age + sex + bmi + smoking + steroids +
                 thyroid + tumor_stage + ecog_score_cat2 + Paclitaxel +
                 Pemetrexed + Gemcitabine + ici + score +
                 pre_HGB_45days + pre_PLT_45days,
               data = vif_dat, family = binomial)

vif_res <- vif(vif_mod)
if (is.matrix(vif_res)) {
  vif_res %>% as.data.frame() %>% rownames_to_column("term") %>%
    arrange(desc(`GVIF^(1/(2*Df))`)) %>% knitr::kable(digits = 2, caption = "VIF — eGFR ratio model (whole cohort)")
} else {
  tibble(term = names(vif_res), VIF = vif_res) %>%
    arrange(desc(VIF)) %>% knitr::kable(digits = 2, caption = "VIF — eGFR ratio model (whole cohort)")
}
