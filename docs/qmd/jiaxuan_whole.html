3 Fine-Gray Models — eGFR Ratio (CR-CYS / CG)

egfr_ratio = ckd_epi_gfr_cre_cys_unindex / cockcroft, entered as a linear continuous predictor in a Fine-Gray competing-risk model (death is the competing event).

Adjuster set (per project spec): age, sex, score (Charlson), ecog_score_cat2, cockcroft (CG-CrCl), cancer_type_cat_7, tumor_stage, platin_group (cis vs carbo), high_dose (platinum dose high vs low), Paclitaxel, Pemetrexed, Gemcitabine (platinum partners), bmi, smoking, thyroid. Anemia models additionally adjust for pre_HGB_45days; thrombocytopenia for pre_PLT_45days; AKI and composite models do not add baseline labs.

Code 
fg_model <- function(time_var, status_var, pred_var, pre_labs = NULL) {
  adj <- c("age","sex","score","ecog_score_cat2","cockcroft",
           "cancer_type_cat_7","tumor_stage","platin_group","high_dose",
           "Paclitaxel","Pemetrexed","Gemcitabine",
           "bmi","smoking","thyroid")
  # Drop cockcroft from adjusters when it (or its rescaled version) is the predictor
  if (pred_var %in% c("cockcroft","cockcroft_per10")) {
    adj <- setdiff(adj, "cockcroft")
  }
  if (!is.null(pre_labs)) adj <- c(adj, pre_labs)
  fmla <- as.formula(paste0(
    "Surv(", time_var, ", ", status_var, ") ~ ",
    paste(c(pred_var, adj), collapse = " + ")
  ))
  tidycmprsk::crr(fmla, failcode = 1, data = dat) %>%
    tbl_regression(exp = TRUE,
                   pvalue_fun = function(x) style_pvalue(x, digits = 3)) %>%
    add_n()
}

4 Fine-Gray Models — Cockcroft-Gault eGFR (per 10 mL/min)

Same adjuster set as Section 3, except the continuous predictor is cockcroft_per10 (= cockcroft / 10). The reported HR is per 10 mL/min higher Cockcroft-Gault eGFR — i.e., HR < 1 means higher kidney function is associated with fewer events. Per-1-mL/min HRs are clinically uninterpretable; per-10 is the convention in nephro-oncology (bio_agent confirmed). p-values and model fit are identical to a per-1-unit fit.

6 subHR Curves — eGFR Ratio

Spline-based subHR curves for each competing outcome, estimated from Fine-Gray competing-risk models fit with a natural cubic spline (3 df) on egfr_ratio, adjusting for the same covariate set as Section 3. Curves show the subdistribution hazard ratio (on the original scale) relative to the median observed ratio (dotted vertical line); reference value subHR = 1 (dashed horizontal line). Ribbons are 95% pointwise confidence intervals from the delta method using the spline vcov block. Only the 2nd–98th percentile range of ratio is plotted to limit influence of tails.

Code 
suppressPackageStartupMessages(library(splines))

plot_subhr <- function(time_var, status_var, extra_adj = NULL, outcome_label) {
  base_adj <- c("age","sex","score","ecog_score_cat2","cockcroft",
                "cancer_type_cat_7","tumor_stage","platin_group","high_dose",
                "Paclitaxel","Pemetrexed","Gemcitabine",
                "bmi","smoking","thyroid")
  keep <- c("egfr_ratio", time_var, status_var, base_adj, extra_adj)
  d <- dat %>% dplyr::select(all_of(keep)) %>% tidyr::drop_na()

  ns_b <- ns(d$egfr_ratio, df = 3)
  knots_in <- attr(ns_b, "knots")
  bdry     <- attr(ns_b, "Boundary.knots")
  colnames(ns_b) <- c("ns1","ns2","ns3")

  adj_df <- d[, c(base_adj, extra_adj), drop = FALSE]
  X_adj <- model.matrix(~ ., data = adj_df)[, -1, drop = FALSE]
  X <- cbind(as.matrix(ns_b), X_adj)

  stat_num <- suppressWarnings(as.integer(as.character(d[[status_var]])))
  time_num <- as.numeric(d[[time_var]])

  fit <- cmprsk::crr(ftime = time_num, fstatus = stat_num,
                     cov1 = X, failcode = 1)

  grid_x <- seq(quantile(d$egfr_ratio, 0.02, na.rm = TRUE),
                quantile(d$egfr_ratio, 0.98, na.rm = TRUE),
                length.out = 100)
  ref_x  <- median(d$egfr_ratio, na.rm = TRUE)

  ns_grid <- predict(ns(d$egfr_ratio, knots = knots_in, Boundary.knots = bdry),
                     grid_x)
  ns_ref  <- predict(ns(d$egfr_ratio, knots = knots_in, Boundary.knots = bdry),
                     ref_x)
  ns_diff <- sweep(as.matrix(ns_grid), 2, as.vector(ns_ref), "-")

  coef_ns <- fit$coef[1:3]
  V_ns    <- fit$var[1:3, 1:3]

  log_hr <- as.vector(ns_diff %*% coef_ns)
  se     <- sqrt(pmax(diag(ns_diff %*% V_ns %*% t(ns_diff)), 0))

  df_plot <- tibble(
    ratio = grid_x,
    subhr = exp(log_hr),
    lcl   = exp(log_hr - 1.96 * se),
    ucl   = exp(log_hr + 1.96 * se)
  )

  ggplot(df_plot, aes(x = ratio, y = subhr)) +
    geom_ribbon(aes(ymin = lcl, ymax = ucl), alpha = 0.2, fill = "steelblue") +
    geom_line(color = "steelblue", linewidth = 1) +
    geom_hline(yintercept = 1, linetype = "dashed", color = "grey50") +
    geom_vline(xintercept = ref_x, linetype = "dotted", color = "grey40") +
    labs(x = "CKD-EPI Cr-Cys / CG ratio",
         y = "subHR (95% CI)",
         title = outcome_label) +
    coord_cartesian(xlim = c(0.5, 1.5), ylim = c(0, 4)) +
    theme_minimal(base_size = 11)
}

7 Linear subHR Curves — eGFR Ratio (raw scale)

Same Fine-Gray models as Section 3 (linear continuous egfr_ratio, no spline) and the same adjuster set, but visualised as a curve. The reference is the median observed ratio (dotted vertical line); subHR(x) = exp(beta · (x − reference)). Because the predictor is linear, the curve is exponential on the original scale and a straight line on the log scale. The 95% CI band widens linearly with distance from the reference. Plotted over the 2nd–98th percentile range of ratio.

Code 
plot_subhr_linear <- function(time_var, status_var, extra_adj = NULL,
                              outcome_label, x_axis = c("ratio","percentile")) {
  x_axis <- match.arg(x_axis)
  base_adj <- c("age","sex","score","ecog_score_cat2","cockcroft",
                "cancer_type_cat_7","tumor_stage","platin_group","high_dose",
                "Paclitaxel","Pemetrexed","Gemcitabine",
                "bmi","smoking","thyroid")
  keep <- c("egfr_ratio", time_var, status_var, base_adj, extra_adj)
  d <- dat %>% dplyr::select(all_of(keep)) %>% tidyr::drop_na()

  adj_df <- d[, c(base_adj, extra_adj), drop = FALSE]
  X_adj <- model.matrix(~ ., data = adj_df)[, -1, drop = FALSE]
  X <- cbind(egfr_ratio = d$egfr_ratio, X_adj)

  stat_num <- suppressWarnings(as.integer(as.character(d[[status_var]])))
  time_num <- as.numeric(d[[time_var]])

  fit <- cmprsk::crr(ftime = time_num, fstatus = stat_num,
                     cov1 = X, failcode = 1)

  beta   <- fit$coef[1]
  V_beta <- fit$var[1, 1]

  ref_x  <- median(d$egfr_ratio, na.rm = TRUE)
  grid_x <- seq(quantile(d$egfr_ratio, 0.02, na.rm = TRUE),
                quantile(d$egfr_ratio, 0.98, na.rm = TRUE),
                length.out = 100)

  log_hr <- beta * (grid_x - ref_x)
  se     <- sqrt((grid_x - ref_x)^2 * V_beta)

  if (x_axis == "percentile") {
    ecdf_fn      <- ecdf(d$egfr_ratio)
    grid_display <- ecdf_fn(grid_x) * 100
    ref_display  <- ecdf_fn(ref_x)  * 100
    xlab <- "Percentile of CKD-EPI Cr-Cys / CG ratio"
  } else {
    grid_display <- grid_x
    ref_display  <- ref_x
    xlab <- "CKD-EPI Cr-Cys / CG ratio"
  }

  df_plot <- tibble(
    x     = grid_display,
    subhr = exp(log_hr),
    lcl   = exp(log_hr - 1.96 * se),
    ucl   = exp(log_hr + 1.96 * se)
  )

  ggplot(df_plot, aes(x = x, y = subhr)) +
    geom_ribbon(aes(ymin = lcl, ymax = ucl), alpha = 0.2, fill = "darkorange") +
    geom_line(color = "darkorange", linewidth = 1) +
    geom_hline(yintercept = 1, linetype = "dashed", color = "grey50") +
    geom_vline(xintercept = ref_display, linetype = "dotted", color = "grey40") +
    labs(x = xlab, y = "subHR (95% CI)", title = outcome_label) +
    coord_cartesian(
      xlim = if (x_axis == "ratio") c(0.5, 1.5) else NULL,
      ylim = c(0, 4)
    ) +
    theme_minimal(base_size = 11)
}
