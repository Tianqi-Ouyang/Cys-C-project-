# Create eGFR ratio: CKD-EPI Cr-Cys (unindexed) / Cockcroft-Gault
# Also create per-10 mL/min rescaled versions of cockcroft and CKD-EPI for
# reporting (HRs per 1 mL/min are clinically uninterpretable; convention in
# nephro-onc is per 10 mL/min — bio_agent confirmed). Adjusters keep raw scale.
dat <- dat %>%
  mutate(
    # Ratio uses UNCAPPED eGFR
    egfr_ratio    = ckd_epi_gfr_cre_cys_unindex / cockcroft,
    egfr_ratio_per10 = -egfr_ratio * 10,  # per 0.1-unit *decrease* reporting scale
    egfr_discrepancy = ckd_epi_gfr_cre_cys_unindex - cockcroft,
    cockcroft_per10 = cockcroft / 10,
    ckd_epi_per10   = ckd_epi_gfr_cre_cys_unindex / 10
  )

3 Fine-Gray Models — eGFR Ratio (adjusted for Cockcroft-Gault)

fg_model <- function(time_var, status_var, pred_var, pre_labs = NULL,
                     egfr_adj = c("none", "cockcroft", "ckd_epi")) {
  egfr_adj <- match.arg(egfr_adj)
  adj <- c("age","sex","score","ecog_score_cat2",
           "cancer_type_cat_7","tumor_stage","platin_group","high_dose",
           "Paclitaxel","Pemetrexed","Gemcitabine","ici",
           "bmi","smoking","steroids","thyroid")
  # Drop platin_group when predictor requires dose data (carbo-only after drop_na)
  if (grepl("dose_discrep", pred_var)) adj <- setdiff(adj, "platin_group")
  if (egfr_adj == "cockcroft") adj <- c(adj, "cockcroft")
  if (egfr_adj == "ckd_epi")   adj <- c(adj, "ckd_epi_gfr_cre_cys_unindex")
  if (!is.null(pre_labs)) adj <- c(adj, pre_labs)
  fmla <- as.formula(paste0(
    "Surv(", time_var, ", ", status_var, ") ~ ",
    paste(c(pred_var, adj), collapse = " + ")
  ))
  tidycmprsk::crr(fmla, failcode = 1, data = dat) %>%
    tbl_regression(exp = TRUE,
                   pvalue_fun = function(x) style_pvalue(x, digits = 3)) %>%
    add_n()
}

cox_model <- function(time_var, status_var, pred_var, pre_labs = NULL,
                      egfr_adj = c("none", "cockcroft", "ckd_epi")) {
  egfr_adj <- match.arg(egfr_adj)
  adj <- c("age","sex","score","ecog_score_cat2",
           "cancer_type_cat_7","tumor_stage","platin_group","high_dose",
           "Paclitaxel","Pemetrexed","Gemcitabine","ici",
           "bmi","smoking","steroids","thyroid")
  if (grepl("dose_discrep", pred_var)) adj <- setdiff(adj, "platin_group")
  if (egfr_adj == "cockcroft") adj <- c(adj, "cockcroft")
  if (egfr_adj == "ckd_epi")   adj <- c(adj, "ckd_epi_gfr_cre_cys_unindex")
  if (!is.null(pre_labs)) adj <- c(adj, pre_labs)
  fmla <- as.formula(paste0(
    "Surv(", time_var, ", ", status_var, ") ~ ",
    paste(c(pred_var, adj), collapse = " + ")
  ))
  survival::coxph(fmla, data = dat) %>%
    tbl_regression(exp = TRUE,
                   pvalue_fun = function(x) style_pvalue(x, digits = 3)) %>%
    add_n()
}

4 Cause-Specific Hazard Models — eGFR Ratio (adjusted for Cockcroft-Gault)

csh_model <- function(time_var, status_var, pred_var, pre_labs = NULL) {
  adj <- c("age","sex","score","ecog_score_cat2",
           "cancer_type_cat_7","tumor_stage","platin_group","high_dose",
           "Paclitaxel","Pemetrexed","Gemcitabine","ici",
           "bmi","smoking","steroids","thyroid","cockcroft")
  if (grepl("dose_discrep", pred_var)) adj <- setdiff(adj, "platin_group")
  if (!is.null(pre_labs)) adj <- c(adj, pre_labs)
  # Cause-specific: treat death (status == 2) as censored
  d <- dat
  d$csh_status <- ifelse(d[[status_var]] == 1, 1L, 0L)
  fmla <- as.formula(paste0(
    "Surv(", time_var, ", csh_status) ~ ",
    paste(c(pred_var, adj), collapse = " + ")
  ))
  survival::coxph(fmla, data = d) %>%
    tbl_regression(exp = TRUE,
                   pvalue_fun = function(x) style_pvalue(x, digits = 3)) %>%
    add_n()
}

plot_subhr_linear_ratio <- function(time_var, status_var, extra_adj = NULL,
                                    outcome_label, x_axis = c("ratio","percentile")) {
  x_axis <- match.arg(x_axis)
  base_adj <- c("age","sex","score","ecog_score_cat2","cockcroft",
                "cancer_type_cat_7","tumor_stage","platin_group","high_dose",
                "Paclitaxel","Pemetrexed","Gemcitabine","ici",
                "bmi","smoking","steroids","thyroid")
  keep <- c("egfr_ratio", time_var, status_var, base_adj, extra_adj)
  d <- dat %>% dplyr::select(all_of(keep)) %>% tidyr::drop_na()

  adj_df <- d[, c(base_adj, extra_adj), drop = FALSE]
  X_adj <- model.matrix(~ ., data = adj_df)[, -1, drop = FALSE]
  X <- cbind(egfr_ratio = d$egfr_ratio, X_adj)

  stat_num <- suppressWarnings(as.integer(as.character(d[[status_var]])))
  time_num <- as.numeric(d[[time_var]])

  fit <- cmprsk::crr(ftime = time_num, fstatus = stat_num,
                     cov1 = X, failcode = 1)

  beta   <- fit$coef[1]
  V_beta <- fit$var[1, 1]

  ref_x  <- median(d$egfr_ratio, na.rm = TRUE)
  grid_x <- seq(quantile(d$egfr_ratio, 0.025, na.rm = TRUE),
                quantile(d$egfr_ratio, 0.975, na.rm = TRUE),
                length.out = 100)

  log_hr <- beta * (grid_x - ref_x)
  se     <- sqrt((grid_x - ref_x)^2 * V_beta)

  if (x_axis == "percentile") {
    ecdf_fn      <- ecdf(d$egfr_ratio)
    grid_display <- ecdf_fn(grid_x) * 100
    ref_display  <- ecdf_fn(ref_x)  * 100
    xlab <- "Percentile of eGFR Ratio"
  } else {
    grid_display <- grid_x
    ref_display  <- ref_x
    xlab <- "CKD-EPI Cr-Cys / CG ratio"
  }

  df_plot <- tibble(
    x     = grid_display,
    subhr = exp(log_hr),
    lcl   = exp(log_hr - 1.96 * se),
    ucl   = exp(log_hr + 1.96 * se)
  )

  p <- ggplot(df_plot, aes(x = x, y = subhr)) +
    geom_ribbon(aes(ymin = lcl, ymax = ucl), alpha = 0.2, fill = "darkorange") +
    geom_line(color = "darkorange", linewidth = 1) +
    geom_hline(yintercept = 1, linetype = "dashed", color = "grey50") +
    geom_vline(xintercept = ref_display, linetype = "dotted", color = "grey40") +
    labs(x = xlab, y = "subHR (95% CI)", title = outcome_label) +
    theme_minimal(base_size = 11)

  if (x_axis == "ratio") {
    p <- p + coord_cartesian(xlim = c(0.65, 1.45), ylim = c(0, 4))
  } else {
    p <- p + coord_cartesian(xlim = c(0.65, 1.45), ylim = c(0, 4))
  }
  p
}
