3 Spline subHR Curves by eGFR CR-CYS / CG Ratio (Carboplatin)

egfr_ratio = ckd_epi_gfr_cre_cys_unindex / cockcroft

Each plot shows the subdistribution hazard ratio (subHR, 95% CI) from a Fine-Gray model with a restricted cubic spline (natural spline basis) on the ratio, relative to the median ratio (dotted vertical line). The x-axis is the raw ratio trimmed to the 2.5–97.5 percentiles; the rug at the bottom shows the observed distribution. All models are adjusted for the same covariate set as Section 4.

Code 
library(splines)

subhr_curve <- function(data, time_var, status_var, xvar,
                        pre_lab = NULL, df_spline = NULL,
                        trim = c(0.025, 0.975), ref_fun = median,
                        title = NULL, xlab = "eGFR CR-CYS / CG ratio") {

  adj <- c("age","sex","bmi","smoking","steroids","thyroid","tumor_stage",
           "ecog_score_cat2","Paclitaxel","Pemetrexed","Gemcitabine",
           "ici","score")
  if (!is.null(pre_lab)) adj <- c(adj, pre_lab)

  d <- data %>%
    dplyr::select(dplyr::all_of(c(time_var, status_var, xvar, adj))) %>%
    tidyr::drop_na()

  stat_num <- as.numeric(as.character(d[[status_var]]))

  x <- d[[xvar]]
  n_events <- sum(stat_num == 1, na.rm = TRUE)

  if (is.null(df_spline)) {
    df_spline <- if (n_events < 100) 3 else if (n_events < 200) 4 else 5
  }

  basis <- splines::ns(x, df = df_spline)

  adj_mf <- model.matrix(~ ., data = d[, adj, drop = FALSE])[, -1, drop = FALSE]

  X <- cbind(basis, adj_mf)
  colnames(X) <- make.names(colnames(X), unique = TRUE)

  fit <- cmprsk::crr(
    ftime   = d[[time_var]],
    fstatus = stat_num,
    cov1    = X,
    failcode = 1, cencode = 0
  )

  k <- ncol(basis)
  beta_s <- fit$coef[seq_len(k)]
  V_s    <- fit$var[seq_len(k), seq_len(k), drop = FALSE]

  ref  <- ref_fun(x)
  q_lo <- as.numeric(quantile(x, trim[1], na.rm = TRUE))
  q_hi <- as.numeric(quantile(x, trim[2], na.rm = TRUE))
  x_grid <- seq(q_lo, q_hi, length.out = 200)

  basis_ref  <- predict(basis, newx = ref)
  basis_grid <- predict(basis, newx = x_grid)
  L <- sweep(basis_grid, 2, as.numeric(basis_ref), "-")

  log_shr <- as.numeric(L %*% beta_s)
  se      <- sqrt(diag(L %*% V_s %*% t(L)))

  dfp <- tibble::tibble(
    x   = x_grid,
    shr = exp(log_shr),
    lo  = exp(log_shr - 1.96 * se),
    hi  = exp(log_shr + 1.96 * se)
  )

  rug_df <- tibble::tibble(x = x[x >= q_lo & x <= q_hi])

  ggplot(dfp, aes(x, shr)) +
    geom_ribbon(aes(ymin = lo, ymax = hi), fill = "steelblue", alpha = 0.25) +
    geom_line(color = "steelblue", linewidth = 1.3) +
    geom_hline(yintercept = 1, linetype = "dashed", color = "grey40") +
    geom_vline(xintercept = ref, linetype = "dotted", color = "grey40") +
    geom_rug(data = rug_df, aes(x = x), inherit.aes = FALSE,
             sides = "b", alpha = 0.15) +
    labs(title = title, x = xlab, y = "subHR (95% CI)") +
    theme_minimal(base_size = 12) +
    theme(
      axis.title.x = element_text(size = 11),
      axis.title.y = element_text(size = 11),
      plot.title   = element_text(size = 13, face = "bold")
    )
}

4 Fine-Gray Models — Three Parallel Predictor Sets (Carboplatin)

Three parallel Fine-Gray models are fit per outcome, each using a different continuous kidney-function predictor (the ratio, Cockcroft-Gault alone, and CKD-EPI Cr-Cys unindexed alone). The predictors are not co-entered in a single model because the ratio is their quotient. All three share the same adjuster set: age + sex + bmi + smoking + steroids + thyroid + tumor_stage + ecog_score_cat2 + Paclitaxel + Pemetrexed + Gemcitabine + ici + score, plus pre_HGB_45days (anemia), pre_PLT_45days (thrombocytopenia), pre_CRE_45days (AKI), or both baseline labs (composite).

Code 
fg_tbl <- function(time_var, status_var, pred_var, pre_lab = NULL) {
  adj <- c("age","sex","bmi","smoking","steroids","thyroid","tumor_stage",
           "ecog_score_cat2","Paclitaxel","Pemetrexed","Gemcitabine",
           "ici","score")
  if (!is.null(pre_lab)) adj <- c(adj, pre_lab)
  fmla <- as.formula(paste0(
    "Surv(", time_var, ", ", status_var, ") ~ ",
    paste(c(pred_var, adj), collapse = " + ")
  ))
  tidycmprsk::crr(fmla, failcode = 1, data = dat) %>%
    tbl_regression(exp = TRUE,
                   pvalue_fun = function(x) style_pvalue(x, digits = 3)) %>%
    add_n()
}

fg_three <- function(time_var, status_var, pre_lab = NULL) {
  t1 <- fg_tbl(time_var, status_var, "egfr_ratio",                 pre_lab)
  t2 <- fg_tbl(time_var, status_var, "cockcroft",                  pre_lab)
  t3 <- fg_tbl(time_var, status_var, "ckd_epi_gfr_cre_cys_unindex",pre_lab)
  tbl_merge(
    list(t1, t2, t3),
    tab_spanner = c("**eGFR Ratio (CR-CYS / CG)**",
                    "**Cockcroft-Gault**",
                    "**CKD-EPI Cr-Cys (unindexed)**")
  )
}
