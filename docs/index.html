3.1 standardize_colnames

standardize_colnames <- function(df) {
  # Replace non-alphanumeric characters with underscore
  names(df) <- gsub("[^[:alnum:]_]", "_", names(df))

  # Reduce multiple underscores to a single underscore
  names(df) <- gsub("_+", "_", names(df))

  # Remove underscore if it's at the end of the name
  names(df) <- gsub("_$", "", names(df))

  # Return the cleaned dataframe
  return(df)
}

3.2 create_events_and_times_1_11

Creates event_hosp_1–event_hosp_11 and time_to_event_hosp_1–time_to_event_hosp_11 from hosp_visit_0_* (baseline) and extra hosp_visit_1_*–hosp_visit_5_* rows.

Code 
create_events_and_times_1_11 <- function(data) {
    # Initialize hospital event and time variables
    for (i in 1:11) {
        data[[paste0("event_hosp_", i)]] <- 0
        data[[paste0("time_to_event_hosp_", i)]] <- NA
    }

    # Process each row
    for (i in seq_len(nrow(data))) {
        # If baseline hospital visit is missing, set all to 0/NA
        if (is.na(data[i, "hosp_visit_0_diff"])) {
            for (j in 1:11) {
                data[i, paste0("event_hosp_", j)] <- 0
                data[i, paste0("time_to_event_hosp_", j)] <- NA
            }
        } else {
            # Baseline hospital visit
            for (j in 1:11) {
                if (!is.na(data[i, paste0("hosp_visit_0_reason_", j)]) &&
                    data[i, paste0("hosp_visit_0_reason_", j)] == 1) {
                    data[i, paste0("event_hosp_", j)] <- 1
                    data[i, paste0("time_to_event_hosp_", j)] <- data[i, "hosp_visit_0_diff"]
                }
            }
        }

        # Additional hospital visits
        if (!is.na(data[i, "other_hosp"]) && data[i, "other_hosp"] == 1) {
            extra_hosp_n <- as.numeric(data[i, "extra_hosp_n"])
            if (!is.na(extra_hosp_n) && extra_hosp_n > 0) {
                for (visit in 1:extra_hosp_n) {
                    for (j in 1:11) {
                        if (!is.na(data[i, paste0("event_hosp_", j)]) &&
                            data[i, paste0("event_hosp_", j)] == 0 &&
                            !is.na(data[i, paste0("hosp_visit_", visit, "_reason_", j)]) &&
                            data[i, paste0("hosp_visit_", visit, "_reason_", j)] == 1) {
                            data[i, paste0("event_hosp_", j)] <- 1
                            data[i, paste0("time_to_event_hosp_", j)] <- data[i, paste0("hosp_visit_", visit, "_diff")]
                        }
                    }
                }
            }
        }
    }

    return(data)
}

3.3 create_drug_indicators

create_drug_indicators <- function(data) {

  # Create indicator for Bevacizumab
  data$Bevacizumab <- ifelse(grepl("Bevacizumab|Avastin", data$regimen_dose, ignore.case = TRUE), 1, 0)

  # Create indicator for Paclitaxel
  data$Paclitaxel <- ifelse(grepl("Paclitaxel|Taxol|Abraxane", data$regimen_dose, ignore.case = TRUE), 1, 0)

  # Create indicator for Pemetrexed
  data$Pemetrexed <- ifelse(grepl("Pemetrexed|pem\\b", data$regimen_dose, ignore.case = TRUE), 1, 0)

  # Create indicator for Durvalumab
  data$Durvalumab <- ifelse(grepl("Durvalumab|Durva", data$regimen_dose, ignore.case = TRUE), 1, 0)

  # Create indicator for Doxorubicin
  data$Doxorubicin <- ifelse(grepl("Doxorubicin|Adriamycin", data$regimen_dose, ignore.case = TRUE), 1, 0)

  # Create indicator for Pembrolizumab
  data$Pembrolizumab <- ifelse(grepl("Pembrolizumab|Pembro", data$regimen_dose, ignore.case = TRUE), 1, 0)

  # Create indicator for Etoposide
  data$Etoposide <- ifelse(grepl("Etoposide|etop\\b", data$regimen_dose, ignore.case = TRUE), 1, 0)

  # Create indicator for Gemcitabine
  data$Gemcitabine <- ifelse(grepl("Gemcitabine|\\bGem\\b", data$regimen_dose, ignore.case = TRUE), 1, 0)

  # Create indicator for Atezolizumab
  data$Atezolizumab <- ifelse(grepl("Atezolizumab", data$regimen_dose, ignore.case = TRUE), 1, 0)

  # Create indicator for Pertuzumab
  data$Pertuzumab <- ifelse(grepl("Pertuzumab", data$regimen_dose, ignore.case = TRUE), 1, 0)

  # Create indicator for Trastuzumab
  data$Trastuzumab <- ifelse(grepl("Trastuzumab|Traztuzumab|Transtuzumab|Herceptin",
                                   data$regimen_dose, ignore.case = TRUE), 1, 0)

  # Create indicator for Docetaxel
  data$Docetaxel <- ifelse(grepl("Docetaxel|Taxotere", data$regimen_dose, ignore.case = TRUE), 1, 0)

  # Create indicator for Fluorouracil
  data$Fluorouracil <- ifelse(grepl("Fluorouracil|5FU", data$regimen_dose, ignore.case = TRUE), 1, 0)

  # Create indicator for Methotrexate
  data$Methotrexate <- ifelse(grepl("Methotrexate", data$regimen_dose, ignore.case = TRUE), 1, 0)

  # Create indicator for Vinblastine
  data$Vinblastine <- ifelse(grepl("Vinblastine", data$regimen_dose, ignore.case = TRUE), 1, 0)

  # Create indicator for Administered alone (absence of all the above drugs)
  all_drugs <- c("Bevacizumab", "Paclitaxel", "Pemetrexed", "Durvalumab", "Doxorubicin",
                 "Pembrolizumab", "Etoposide", "Gemcitabine", "Atezolizumab", "Pertuzumab",
                 "Trastuzumab", "Docetaxel", "Fluorouracil", "Methotrexate", "Vinblastine")

  data$Administered_alone <- ifelse(rowSums(data[, all_drugs]) == 0, 1, 0)

  return(data)
}

3.4 cmp_variable

Builds competing-risk time and status for a single outcome. Status coding: 0 = censored, 1 = event occurred, 2 = death (competing event). If death occurs first, the death time and status (0/2) override the event.

Code 
cmp_variable <- function(data, event_time, event_status, name) {
  # Enquote the time and status variables to allow their evaluation within the data context
  event_time <- enquo(event_time)
  event_status <- enquo(event_status)


  # Create dynamic names for the new columns
  cmp_time_name <- paste0("cmp_time_", name)
  cmp_status_name <- paste0("cmp_status_", name)

  # Add new columns to the dataset using mutate()
  data %>%
    mutate(
      !!cmp_time_name := ifelse(time_to_death_90 <= !!event_time ,time_to_death_90,!!event_time),
      !!cmp_status_name := ifelse(time_to_death_90 <= !!event_time,time_to_death_90_status,!!event_status)
    )
}

3.5 pre_lab

Finds the closest lab value on or before index_date within the time window. Returns one row per patient (most recent qualifying lab result).

###### pre lab  index data included closest date
pre_lab <- function(data, month, time, index_date, Group_Id) {
  index_date <- enquo(index_date)
  if (month) {
    time_value <- time * 30
  } else {
    time_value <- time
  }


  lab_name <- paste0("pre_", Group_Id, "_", time_value, "days")
  lab_date_name <- paste0("pre_", Group_Id, "_", time_value, "days_date")

  result <- data %>%
    filter(Group_Id == !!Group_Id, grepl("^\\d+\\.\\d+$", Result)) %>%
    filter(!is.na(Result)) %>%
    mutate(Result = as.numeric(Result)) %>%
    filter(Seq_Date_Time >= (!!index_date) - time_value, Seq_Date_Time <= !!index_date) %>%
    arrange(EMPI, desc(Seq_Date_Time)) %>%
    distinct(EMPI, .keep_all = TRUE) %>%
    rename(!!lab_name := Result,
           !!lab_date_name := Seq_Date_Time) %>%
    select(EMPI,
           !!lab_name, !!lab_date_name)

  return(result)
}

## post lab cutoff function
post_lab_less_than_cutpoint <- function(data,days,index_date,Group_ID,Group_ID_name,outcome_name,cutpoint) {
  index_date <- enquo(index_date)
  lab_name <- paste0("post_", Group_ID_name, "_cutpoint_", days, "days")
  lab_date_name <- paste0("post_", Group_ID_name, "_cutpoint_", days, "days_date")

  result <- data %>%
    filter(Group_Id %in% Group_ID) %>%
    filter(grepl("^[><]?\\d+\\.?\\d* ?$",Result)) %>%
    mutate(Result=as.numeric(parse_number(Result))) %>%
    filter(!is.na(Result)) %>%
    filter(Seq_Date_Time > (!!index_date), Seq_Date_Time <= !!index_date + days) %>%
    filter(Result<cutpoint) %>%
    arrange(EMPI,Seq_Date_Time) %>%
    distinct(EMPI, .keep_all = TRUE) %>%
    rename(!!lab_name := Result,
           !!lab_date_name := Seq_Date_Time) %>%
    select(EMPI,
           !!lab_name, !!lab_date_name)
  return(result)
}

3.7 dia_list_name

Identifies patients with a given diagnosis (and optionally medication) on or before index_date. Supports excluding specific diagnosis name strings.

Code 
dia_list_name <- function(data, listed_names, index_date, excluded_names = NULL, med_data = NULL, med_names = NULL) {
  #
  if (!is.null(excluded_names)) {
    excluded_pattern <- paste(
      gsub("([][()])", "\\\\\\1", excluded_names),  #
      collapse = "|"
    )
  }

  if (is.null(excluded_names)) {
    dia_result <- data %>%
      filter(Date <= {{index_date}}) %>%
      filter(grepl(paste(listed_names, collapse = "|"), Diagnosis_Name, ignore.case = TRUE)) %>%
      distinct(EMPI)
  } else {
    dia_result <- data %>%
      filter(Date <= {{index_date}}) %>%
      filter(
        grepl(paste(listed_names, collapse = "|"), Diagnosis_Name, ignore.case = TRUE),
        !grepl(excluded_pattern, Diagnosis_Name, ignore.case = TRUE)  #
      ) %>%
      distinct(EMPI)
  }

  if (!is.null(med_data)) {
    med_result <- med_data %>%
      filter(Medication_Date <= {{index_date}}) %>%
      filter(grepl(paste(med_names, collapse = "|"), Medication, ignore.case = TRUE)) %>%
      distinct(EMPI)
  } else {
    med_result <- NULL
  }

  final_result <- rbind(dia_result, med_result) %>%
    distinct(EMPI)

  return(final_result)
}

Cockcroft <- function(age, weight, male, cre) {
  if (male == 0) {
    value <- (140 - age) * weight * 0.85 / (72 * cre)
  } else {
    value <- (140 - age) * weight / (72 * cre)
  }
  return(value)
}
Code 
# CKD-EPI 2012 creatinine-based eGFR (race-free CKD-EPI equation)
ckd_epi_gfr <- function(cre, male, age) {
  k <- ifelse(male == 1, 0.9, 0.7)
  a <- ifelse(male == 1, -0.302, -0.241)
  m <- ifelse(male == 1, 1, 1.012)
  gfr <- 142 * min(c(cre/k, 1))^a * max(c(cre/k, 1))^-1.2 * 0.9938^age * m
  return(gfr)
}
Code 
# CKD-EPI 2012 cystatin C-based eGFR (Inker et al. 2012)
# Age coefficient = 0.996 per official NKF/Inker 2012 (not 0.9961, which is the 2021 Cr+CysC value)
ckd_epi_cys_gfr <- function(cys, male, age) {
  gfr <- 133 * min(c(cys/0.8, 1))^(-0.499) * max(c(cys/0.8, 1))^(-1.328) *
         0.996^age * ifelse(male == 0, 0.932, 1)
  return(gfr)
}
Code 
# Combined creatinine + cystatin C eGFR — exact coefficients from R code summary.Rmd
# Note: coefficients differ from CKD-EPI 2021 race-free (a=-0.144/-0.219, max_cr=-0.544,
# min_cys=-0.323, max_cys=-0.778, age=0.9961, female=0.963)
ckd_epi_cre_cys_gfr <- function(creatinine, cystatin_c, male, age) {
  k      <- ifelse(male == 1, 0.9, 0.7)
  a      <- ifelse(male == 1, -0.144, -0.219)
  female <- ifelse(male == 0, 0.963, 1)
  gfr_cr_cys <- 135 *
    min(c(creatinine/k, 1.0))^a  *
    max(c(creatinine/k, 1.0))^(-0.544) *
    min(c(cystatin_c/0.8, 1.0))^(-0.323) *
    max(c(cystatin_c/0.8, 1.0))^(-0.778) *
    0.9961^age * female
  return(gfr_cr_cys)
}

# Adjust indexed eGFR for body surface area
unindex <- function(value, bsa) {
  return(value * bsa / 1.73)
}
